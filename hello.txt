fibonacci:
    mov r0, #10
    cmp r0, #0
    beq zero_case

    cmp r0, #1
    beq one_case

    mov r1, #0
    mov r2, #1

    mov r3, #2

loop:
    cmp r3, r0     
    bgt end_loop    

    add r4, r1, r2    
    mov r1, r2       
    mov r2, r4         

    add r3, r3, #1      

    b loop         

end_loop:
    mov r0, r2      
    b end;

zero_case:
    mov r0, #0         
    b end

one_case:
    mov r0, #1          
    b end

end: 
  b end
